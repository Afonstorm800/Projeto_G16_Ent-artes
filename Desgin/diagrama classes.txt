@startuml
title Diagrama de Classes - Ent'Artes (Versão Formal)

' Remove estilos de cores e sombras para um look limpo
skinparam monochrome true
skinparam shadowing false
skinparam classAttributeIconSize 0

abstract class Utilizador {
    -int id
    -string nome
    -string email
    -string password
    +login()
}

class Professor {
    -string especialidade
    -float valor_hora_base
    +consultarDisponibilidade()
    +confirmarAula()
}

class Encarregado {
    -string nif
    -string telefone
    +requisitarCoaching()
    +visualizarVagas()
    +confirmarConclusao()
}

class Direcao {
    -string cargo
    +validarRequisicao(id_marcacao)
    +validarFinalizacao(id_marcacao)
    +gerirInventario()
    +emitirRelatorioFaturacao()
}

class Aluno {
    -int id
    -string nome
    -date data_nascimento
}

class Estudio {
    -int id_estudio
    -string nome
    -int capacidade_max
    -list modalidades_permitidas
    -boolean disponivel
    +verificarCompatibilidade(modalidade)
}

class MarcacaoCoaching {
    -int id_marcacao
    -datetime data_hora
    -int duracao_minutos
    -Enum formato
    -Enum status
    -float valor_excecional
    +calcularCusto()
    +validarPrazo48h()
}

class ItemInventario {
    -int id_item
    -string categoria
    -string descricao
    -string tamanho
    -float taxa_simbolica
    -boolean disponivel
    +reservar()
}

class Faturacao {
    -int id_fatura
    -month mes_referencia
    -float total_mensal
    +gerarExcel()
}

' Relacionamentos e Herança
Utilizador <|-- Professor
Utilizador <|-- Encarregado
Utilizador <|-- Direcao

Encarregado "1" -- "*" Aluno : Responsável por
MarcacaoCoaching "*" -- "1" Estudio : Ocorre em
MarcacaoCoaching "*" -- "1" Professor : Lecionada por
MarcacaoCoaching "*" -- "1..8" Aluno : Participantes

Direcao "1" -- "*" MarcacaoCoaching : Valida
Direcao "1" -- "*" Faturacao : Consolida
Direcao "1" -- "*" ItemInventario : Gere

Faturacao "1" o-- "*" MarcacaoCoaching : Agrega dados
Faturacao "1" o-- "*" ItemInventario : Inclui taxas

@enduml
